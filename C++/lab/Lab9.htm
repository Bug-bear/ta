  1. Identify the design flaw in this program and correct it.

class Rect

{

public:

        Rect(double, double);

        double getwidth() const;

        double getheight() const;

        void setwidth(double w);

        void setheight(double h);

        double area() const;

private:

        double height;

        double width;

};

 

double calc_area(double& h, double& w)

{

        return h * w;

}

 

Rect::Rect(double h, double w)

{

        height = h;

        width = w;

}

 

double Rect::getwidth() const

{

        return width;

}

 

double Rect::getheight() const

{

        return height;

}

 

void Rect::setwidth(double w)

{

        width = w;

}

 

void Rect::setheight(double h)

{

        height = h;

}

 

double Rect::area() const

{

        return calc_area(height, width);

}
